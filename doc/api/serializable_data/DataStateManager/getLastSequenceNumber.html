Future<int> getLastSequenceNumber(String fromId, String spec) async {
  var c = Completer<int>();

  try {
    var matches = await matchRelationships(fromId: fromId, type: spec);
    int lastSeq = -1;
    for (var item in matches) {
      int seq = item.sequence!;
      lastSeq = lastSeq < seq ? seq : lastSeq;
    }
    c.complete(lastSeq);
  } catch (ex, st) {
    c.completeError('$fromId $spec -  ${ex.toString()}', st);
  }

  return c.future;
}
